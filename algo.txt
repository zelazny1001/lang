import pandas as pd
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

file_path = 'your_file.csv'
data = pd.read_csv(file_path)

x = data['macroPause']
y = data['offset']
z = data['diff']

fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

colors = ['blue' if value == 0 else 'red' for value in z]
scatter = ax.scatter(x, y, z, c=colors, marker='o', s=20)

ax.set_xlabel('macroPause')
ax.set_ylabel('offset')
ax.set_zlabel('diff', labelpad=20)

plt.tight_layout()
plt.show()
