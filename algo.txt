import librosa
import soundfile as sf

def convert_mp3_to_wav(input_mp3, output_wav):
    audio_data, sample_rate = librosa.load(input_mp3, sr=None)  # sr=None keeps the original sample rate
    sf.write(output_wav, audio_data, sample_rate)
    print(f"Conversion completed: {output_wav}")

# Example usage
input_mp3  = "/asr/mp3/declined-card-2.mp3"
output_wav = "/asr/mp3/declined-card-2.wav"
convert_mp3_to_wav(input_mp3, output_wav)
