import java.util.List;
import java.util.LinkedList;

    public static String getMostRecentNChunkGroups(List<Chunk> chunkList, String personID, int n) {
        if (chunkList == null || chunkList.isEmpty() || n <= 0) {
            return "";
        }

        LinkedList<String> recentGroups = new LinkedList<>();
        StringBuilder currentGroup = new StringBuilder();
        String lastPersonID = null;

        for (int i = chunkList.size() - 1; i >= 0; i--) {
            Chunk chunk = chunkList.get(i);

            if (personID.equals(chunk.getPersonID())) {
                if (!personID.equals(lastPersonID)) {
                    if (recentGroups.size() < n && currentGroup.length() > 0) {
                        recentGroups.addFirst(currentGroup.toString().trim());
                        currentGroup = new StringBuilder();
                    }
                    currentGroup.append(chunk.getTranscript());
                    if (i > 0) {
                        currentGroup.append(" ");
                    }
                } else {
                    currentGroup.insert(0, chunk.getTranscript() + (currentGroup.length() > 0 ? " " : ""));
                }
            } else if (personID.equals(lastPersonID) && currentGroup.length() > 0) {
                if (recentGroups.size() < n) {
                    recentGroups.addFirst(currentGroup.toString().trim());
                    currentGroup = new StringBuilder();
                }
            }
            lastPersonID = chunk.getPersonID();
        }

        if (personID.equals(lastPersonID) && recentGroups.size() < n && currentGroup.length() > 0) {
            recentGroups.addFirst(currentGroup.toString().trim());
        }

        return String.join(" ", recentGroups);
    }
