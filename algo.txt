import com.fasterxml.jackson.databind.JsonNode;
import java.util.List;

public class JsonUtil {

    public static boolean isPreexisting(JsonNode node, List<JsonNode> priors) {
        if (node == null || priors == null) {
            return false;
        }

        String entityConcept = node.has("entityConcept") ? node.get("entityConcept").asText() : null;
        String rawText = node.has("rawText") ? node.get("rawText").asText() : null;
        String start = node.has("start") ? node.get("start").asText() : null;
        String stop = node.has("stop") ? node.get("stop").asText() : null;

        return priors.stream().anyMatch(prior -> {
            boolean sameEntityConcept = entityConcept != null && entityConcept.equals(prior.path("entityConcept").asText());
            boolean sameRawText = rawText != null && rawText.equals(prior.path("rawText").asText());
            boolean sameStart = start != null && start.equals(prior.path("start").asText());
            boolean sameStop = stop != null && stop.equals(prior.path("stop").asText());

            return sameEntityConcept && sameRawText && sameStart && sameStop;
        });
    }
}
