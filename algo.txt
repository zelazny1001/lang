import com.fasterxml.jackson.annotation.JsonInclude;
import com.fasterxml.jackson.annotation.JsonInclude.Include;

@JsonInclude(Include.NON_NULL)
public class Configuration {
    private static String agentID;
    private static String customerID;
    private static Boolean customerOnlyPurpose;

    public static String getAgentID() {
        return agentID;
    }

    public static void setAgentID(String agentID) {
        Configuration.agentID = agentID;
    }

    public static String getCustomerID() {
        return customerID;
    }

    public static void setCustomerID(String customerID) {
        Configuration.customerID = customerID;
    }

    public static Boolean isCustomerOnlyPurpose() {
        return customerOnlyPurpose;
    }

    public static void setCustomerOnlyPurpose(Boolean customerOnlyPurpose) {
        Configuration.customerOnlyPurpose = customerOnlyPurpose;
    }
}

import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class ConfigController {

    @PostMapping("/config")
    public ResponseEntity<Configuration> updateConfiguration(@RequestBody Configuration newConfig) {
        if (newConfig.getAgentID() != null) {
            Configuration.setAgentID(newConfig.getAgentID());
        }
        if (newConfig.getCustomerID() != null) {
            Configuration.setCustomerID(newConfig.getCustomerID());
        }
        if (newConfig.isCustomerOnlyPurpose() != null) {
            Configuration.setCustomerOnlyPurpose(newConfig.isCustomerOnlyPurpose());
        }

        // Creating a new Configuration object with current settings to return
        Configuration currentConfig = new Configuration();
        currentConfig.setAgentID(Configuration.getAgentID());
        currentConfig.setCustomerID(Configuration.getCustomerID());
        currentConfig.setCustomerOnlyPurpose(Configuration.isCustomerOnlyPurpose());

        return ResponseEntity.ok(currentConfig);
    }
}
