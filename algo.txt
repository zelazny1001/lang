package utils;

import java.util.Arrays;
import java.util.List;

public class BooleanListProcessor {
    List<Boolean> list;
    public BooleanListProcessor(List<Boolean> list) {
        this.list = list;
    }
    
    public int numTrue() {
        list.stream().filter(Boolean::booleanValue).count();
    }
    
    public int numFalse() {
        return list.size() - numTrue();
    }
    
    @Override
    public String toString () {
        return "accepted: " + numTrue() + ", rejected: " + numFalse();
    }
}
