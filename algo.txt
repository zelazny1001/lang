import com.fasterxml.jackson.databind.JsonNode;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class JsonNodeListComparator {

    public static boolean compareJsonNodeLists(List<JsonNode> list1, List<JsonNode> list2) {
        if (list1.size() != list2.size()) {
            return false;
        }

        Map<JsonNode, Integer> map1 = createFrequencyMap(list1);
        Map<JsonNode, Integer> map2 = createFrequencyMap(list2);

        return map1.equals(map2);
    }

    private static Map<JsonNode, Integer> createFrequencyMap(List<JsonNode> list) {
        Map<JsonNode, Integer> frequencyMap = new HashMap<>();
        for (JsonNode node : list) {
            frequencyMap.put(node, frequencyMap.getOrDefault(node, 0) + 1);
        }
        return frequencyMap;
    }

    // main method for testing
    public static void main(String[] args) {
        // Example usage
        // List<JsonNode> list1 = ...
        // List<JsonNode> list2 = ...
        // boolean isEqual = compareJsonNodeLists(list1, list2);
    }
}
