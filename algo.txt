    public static void writeCSVToG711(String inputFilename, String outputFilename) {
        try {
            StringBuilder g711Content = new StringBuilder();

            Files.lines(Paths.get(inputFilename)).forEach(line -> {
                String[] values = line.split(",");
                for (String value : values) {
                    g711Content.append(Byte.parseByte(value.trim())).append(",");
                }
            });

            String[] allValues = g711Content.toString().split(",");
            byte[] g711Data = new byte[allValues.length];
            for (int i = 0; i < allValues.length; i++) {
                g711Data[i] = Byte.parseByte(allValues[i].trim());
            }

            try (FileOutputStream fos = new FileOutputStream(outputFilename)) {
                fos.write(g711Data);
            }
        } catch (IOException e) {
            System.out.println("Error writing G711 file: " + e.getMessage());
        }
    }
