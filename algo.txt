import pandas as pd
import openpyxl
from openpyxl.styles import Font, PatternFill, Border, Side

def compareStrategies(dir, filename):
    # Define paths
    input_excel_filename = dir + filename
    output_excel_filename = dir + filename

    # Load the workbook and the 'output' worksheet using pandas
    output_df = pd.read_excel(input_excel_filename, sheet_name='output')

    # Process the data
    analysis_data = []
    transcripts = output_df.groupby('transcript')
    for transcript, group in transcripts:
        if group['agreePurpose'].eq('NA').any() or group['agreeIntent'].eq('NA').any():
            continue

        if len(group) != 3:
            continue

        analysis_row = {'call': group['sessionID'].str.split('-').str[0].iloc[0]}
        for _, row in group.iterrows():
            strategy = row['strategy']
            prefix = strategy.replace('from', '').replace('3', 'Three')  # 'Origin', 'Reset', 'ThreeBranch'
            analysis_row.update({
                f'{prefix}': row['utterance'],
                f'{prefix}Purpose': row['callPurpose'],
                f'{prefix}Intent': row['rawPurposeIntent'],
                f'agree{prefix}': row['agreePurpose']
            })
        analysis_data.append(analysis_row)

    # Create a DataFrame for the 'analysis' data
    analysis_df = pd.DataFrame(analysis_data)

    # Write to Excel using ExcelWriter
    with pd.ExcelWriter(output_excel_filename, engine='openpyxl') as writer:
        output_df.to_excel(writer, sheet_name='output', index=False)
        analysis_df.to_excel(writer, sheet_name='analysis', index=False)

        # Apply formatting using openpyxl
        workbook = writer.book
        analysis_sheet = workbook['analysis']

        # Set the headers
        for cell in analysis_sheet[1]:
            cell.font = Font(bold=True)
        analysis_sheet.freeze_panes = 'A2'

        # Set cell styles
        font = Font(size=8)
        fill = PatternFill(start_color='90EE90', end_color='90EE90', fill_type='solid')
        thin_border = Border(left=Side(style='dotted'), right=Side(style='dotted'),
                             top=Side(style='dotted'), bottom=Side(style='dotted'))

        # Apply formatting and check for intent agreement
        for row in analysis_sheet.iter_rows(min_row=2):
            for cell in row:
                cell.border = thin_border
                cell.font = font

            if row[7].value == row[8].value == row[9].value:  # Check for matching intents
                for cell in row:
                    cell.fill = fill

    # End of process message
    print("Process completed. Analysis worksheet has been updated.")
