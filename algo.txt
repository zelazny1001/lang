def createCallGroupNames(numberOfGroups=5, suffix='suffix', ext='xlsx'):
    groupNames = []
    for i in range(numberOfGroups):
        groupName = f'{i}_1_{i+1}_0_{suffix}.{ext}'
        groupNames.append(groupName)
    return groupNames

def createCallNames(prefix='Call', groupName=''):
    baseNumber = int(groupName.split('_')[0])
    callNames = []
    for i in range(10):
        currentNumber = baseNumber if i < 9 else baseNumber + 1
        nextDigit = i + 1 if i < 9 else 0
        callName = f'{prefix}_{currentNumber}_{nextDigit}'
        callNames.append(callName)
    return callNames
