import java.io.File;
import java.io.IOException;
import java.nio.file.Path;
import java.nio.file.Paths;

public class PathChecker {

    public static Path getCanonicalPath(String filePath) throws IOException {
        File file = new File(filePath);
        return Paths.get(file.getCanonicalPath());
    }

    public static boolean isValidPath(String parent, String target) {
        try {
            Path parentPath = getCanonicalPath(parent);
            Path targetPath = getCanonicalPath(target);
            return targetPath.startsWith(parentPath);
        } catch (IOException e) {
            e.printStackTrace();
            return false;
        }
    }

    public static void main(String[] args) {
        System.out.println(isValidPath("/grandparent/parent", "parent/child/grandchild")); // false
        System.out.println(isValidPath("/tmp/records", "/tmp/records/location1/sublocation/rec.txt")); // true
        System.out.println(isValidPath("/tmp/records/../records", "/tmp/records/location1/sublocation/rec.txt")); // true
    }
}
