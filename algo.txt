import org.apache.commons.collections4.queue.CircularFifoQueue;
import java.util.Collections;
import java.util.List;
import java.util.ArrayList;

public class SessionIDHistory {

    private CircularFifoQueue<String> sessionIDQueue;

    public SessionIDHistory(int capacity) {
        sessionIDQueue = new CircularFifoQueue<>(capacity);
    }

    public String addSessionID(String sessionID) { // return removed sessionID if any
        if (sessionIDQueue.contains(sessionID)) {
            System.out.println(sessionID + " already present");
            return null;
        }
        String removedSessionID = sessionIDQueue.isAtFullCapacity() ? sessionIDQueue.peek() : null;
        sessionIDQueue.add(sessionID);
        System.out.println("removed: " + sessionID);
        return removedSessionID;
    }

    public void clearSessionIDQueue() {
        sessionIDQueue.clear();
    }

    public List<String> reverseListOfSessions() {
        List<String> reversedList = new ArrayList<>(sessionIDQueue);
        Collections.reverse(reversedList);
        return reversedList;
    }

    public void printSessionIDs() {
        System.out.println("Current sessionIDs in history:");
        for (String sessionID : sessionIDQueue) {
            System.out.println(sessionID);
        }
    }

    public static void main(String[] args) { // sanity check
        SessionIDHistory history = new SessionIDHistory(5); // Only the last 5 sessionIDs will be kept

        String removed = null;
        removed = history.addSessionID("SessionID 1");
        removed = history.addSessionID("SessionID 2");
        removed = history.addSessionID("SessionID 3");
        removed = history.addSessionID("SessionID 4");
        removed = history.addSessionID("SessionID 5");

        removed = history.addSessionID("SessionID 6");
        System.out.println("Removed sessionID: " + removed);

        history.printSessionIDs();

        // Clearing the sessionIDQueue
        history.clearSessionIDQueue();

        history.addSessionID("SessionID 7");
        history.addSessionID("SessionID 8");
        history.addSessionID("SessionID 9");
        history.addSessionID("SessionID 10");
        
        List<String> reversedSessions = history.reverseListOfSessions();
        System.out.println("Reversed sessionIDs: " + reversedSessions);
    }
}
