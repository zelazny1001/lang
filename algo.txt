import pandas as pd
import time
from openpyxl.styles import Font, Border, Side, PatternFill

def compareStrategies(dir, filename):
    input_excel_filename = f"{dir}{filename}"
    df_output = pd.read_excel(input_excel_filename, sheet_name='output')

    analysis_columns = [
        'callID', 'person', 'transcript',
        'fromOrigin', 'originPurpose', 'originIntent', 'agreeOriginPurpose', 'agreeOriginIntent',
        'fromReset', 'resetPurpose', 'resetIntent', 'agreeResetPurpose', 'agreeResetIntent',
        '3Branch', 'branch', '3BranchPurpose', '3BranchIntent', 'agree3BranchPurpose', 'agree3BranchIntent'
    ]

    df_analysis = pd.DataFrame(columns=analysis_columns)
    total_rows = len(df_output)
    for index, row in df_output.iterrows():       
        # Find the other rows with the same 'transcript'
        matching_rows = df_output[df_output['transcript'] == row['transcript']]
        print(f"Processing row {index + 1} of {total_rows}, #matching_rows: {len(matching_rows)}")
        
        # Initialize a dictionary to hold the data for the current 'transcript'
        analysis_row = {col: None for col in analysis_columns}
        analysis_row['transcript'] = row['transcript']
        analysis_row['person'] = row['person']
        analysis_row['callID'] = row['sessionID'].split('-')[0]

        # Fill in the data for each strategy
        for _, strategy_row in matching_rows.iterrows():
            strategy = strategy_row['strategy']
            if strategy == 'fromOrigin':
                analysis_row['fromOrigin'] = strategy_row['utterance']
                analysis_row['originPurpose'] = strategy_row['callPurpose']
                analysis_row['agreeOriginPurpose'] = strategy_row['agreePurpose']
                analysis_row['originIntent'] = strategy_row['rawPurposeIntent']
                analysis_row['agreeOriginIntent'] = strategy_row['agreeIntent']
            elif strategy == 'fromReset':
                analysis_row['fromReset'] = strategy_row['utterance']
                analysis_row['resetPurpose'] = strategy_row['callPurpose']
                analysis_row['agreeResetPurpose'] = strategy_row['agreePurpose']
                analysis_row['resetIntent'] = strategy_row['rawPurposeIntent']
                analysis_row['agreeResetIntent'] = strategy_row['agreeIntent']
            elif strategy == '3Branch':
                analysis_row['3Branch'] = strategy_row['utterance']
                analysis_row['3BranchPurpose'] = strategy_row['callPurpose']
                analysis_row['agree3BranchPurpose'] = strategy_row['agreePurpose']
                analysis_row['3BranchIntent'] = strategy_row['rawPurposeIntent']
                analysis_row['agree3BranchIntent'] = strategy_row['agreeIntent']

        # Add the row to the DataFrame
        df_analysis = pd.concat([df_analysis, pd.DataFrame([analysis_row])], ignore_index=True)

    output_excel_filename = f"{dir}analysis.xlsx"
    with pd.ExcelWriter(output_excel_filename, engine='openpyxl') as writer:
        df_analysis.to_excel(writer, sheet_name='analysis', index=False)

        ws = writer.sheets['analysis']
        
        thin_border = Border(left=Side(style='dotted'), 
                             right=Side(style='dotted'), 
                             top=Side(style='dotted'), 
                             bottom=Side(style='dotted'))

        for row in ws.iter_rows(min_row=1, max_row=len(df_analysis) + 1, min_col=1, max_col=len(analysis_columns)):
            for cell in row:
                cell.border = thin_border
                cell.font = Font(size=8)
#                if cell.row == 1:
#                    cell.font = Font(bold=True)

        ws.freeze_panes = 'A2'

#        for col in ws.columns:
#            cell = col[0]
#            cell.auto_filter.ref = ws.dimensions

        green_fill = PatternFill(start_color='90EE90', end_color='90EE90', fill_type='solid')
        blue_fill = PatternFill(start_color='ADD8E6', end_color='ADD8E6', fill_type='solid')

        for row in ws.iter_rows(min_row=2, max_row=len(df_analysis) + 1, min_col=1, max_col=len(analysis_columns)):
            originIntent = row[5].value
            resetIntent = row[10].value
            threeBranchIntent = row[15].value

            # Check if 'originIntent', 'resetIntent', and '3BranchIntent' are the same and not empty strings
            if originIntent == resetIntent == threeBranchIntent and originIntent != '':
                for cell in row:
                    cell.fill = green_fill

            # Highlight rows where 'person' == 'person2'
