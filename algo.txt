import java.util.ArrayList;
import java.util.List;

public class Main {
    public static void main(String[] args) {
        List<byte[]> list = new ArrayList<>();
        list.add(new byte[]{1, 2, 3});
        list.add(new byte[]{4, 5, 6});
        list.add(new byte[]{7, 8, 9});

        byte[] flattenedArray = flatten(list);

        System.out.println(java.util.Arrays.toString(flattenedArray));
    }

    public static byte[] flatten(List<byte[]> list) {
        int totalLength = list.size() * list.get(0).length;
        byte[] result = new byte[totalLength];
        int currentIndex = 0;

        for (byte[] array : list) {
            System.arraycopy(array, 0, result, currentIndex, array.length);
            currentIndex += array.length;
        }

        return result;
    }
}
