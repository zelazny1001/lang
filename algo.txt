    public static List<TrainDataFeaturized> featurizeData(String featuresPath) {
        return getIntentTrainingData(featuresPath).stream()
                .map(data -> {
                    TrainDataFeaturized featurized = new TrainDataFeaturized();
                    featurized.setUtterance(data.getUtterance());
                    featurized.setIntent(data.getIntent());
                    featurized.setFeatures(data.getFeatures());
                    return featurized;
                })
                .collect(Collectors.toList());
    }
