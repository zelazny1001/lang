package utils;

public class ResultsAnalyzer {

    public static void main(String[] args) {
        int numAccepted = 500;
        for (float offset = 0.5f; offset <= 2.5f; offset += 0.1f) {
            String name = generateName(offset, numAccepted);
            getResults(offset, name);
        }
    }

    public static String generateName(float offset, int numAccepted) {
        return String.format("accepted_voice_%.1f_%d", offset, numAccepted);
    }

    public static void getResults(float offset, String name) {
        System.out.println("Offset: " + offset + ", Name: " + name);
    }
}
