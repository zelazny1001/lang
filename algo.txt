function filterAndSortRows(rows, key) {
    const keyParts = key.split("_"); // Break the key into parts (e.g., ['max', '1218am', 'max', 'customerside8Hz'])

    return rows
        .filter(row => {
            const filename = row.filename.toLowerCase(); // Case-insensitive match
            return keyParts.some(part => filename.includes(part.toLowerCase())); // Check if any part matches
        })
        .sort((a, b) => a.seqNum - b.seqNum); // Sort by seqNum
}
