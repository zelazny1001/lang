class Candidate:
    start_idx: int
    start_dB: float
    num_voice_chunks: int = 0
    num_non_voice_chunks: int = 0

    def add_voice(self):
        self.num_voice_chunks += 1

    def add_non_voice(self):
        self.num_non_voice_chunks += 1

class Span:
    start_idx: int
    end_idx: int
    start_dB: float
    end_dB: float
    num_voice_chunks: int
    num_non_voice_chunks: int
