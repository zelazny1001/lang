import com.fasterxml.jackson.databind.JsonNode;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

public class EntityFilter {

    public String filterForIntentConcepts(String intent, List<JsonNode> entities, CompatibleEntityMapping compatibleEntities) {
        List<String> filteredEntities = new ArrayList<>();

        Map<String, List<String>> mapping = compatibleEntities.getCompatibleEntities();
        List<String> compatibleConcepts = mapping.get(intent);

        if (compatibleConcepts != null) {
            for (JsonNode entity : entities) {
                String entityConcept = entity.get("entityConcept").asText();
                if (compatibleConcepts.contains(entityConcept)) {
                    filteredEntities.add(entity.toString());
                }
            }
        }
        return filteredEntities.toString();
    }
}
