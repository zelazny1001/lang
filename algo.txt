import org.apache.commons.collections4.queue.CircularFifoQueue;

public class MessageHistory {

    private CircularFifoQueue<String> messageQueue;

    public MessageHistory(int capacity) {
        // Initialize the CircularFifoQueue with the given capacity
        messageQueue = new CircularFifoQueue<>(capacity);
    }

    public String addMessage(String message) {
        // Check if adding a new message will remove an old one
        String removedMessage = messageQueue.isAtFullCapacity() ? messageQueue.peek() : null;
        // Add the new message to the queue
        messageQueue.add(message);
        // Return the message that was removed, if any
        return removedMessage;
    }

    public void printMessages() {
        System.out.println("Current messages in history:");
        for (String message : messageQueue) {
            System.out.println(message);
        }
    }

    public static void main(String[] args) {
        MessageHistory history = new MessageHistory(5); // Only the last 5 messages will be kept

        history.addMessage("Message 1");
        history.addMessage("Message 2");
        history.addMessage("Message 3");
        history.addMessage("Message 4");
        history.addMessage("Message 5");

        // At this point, the queue is full, and the next addition will remove the oldest message
        String removed = history.addMessage("Message 6");
        System.out.println("Removed message: " + removed);

        // Print the current messages
        history.printMessages();
    }
}
