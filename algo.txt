import com.fasterxml.jackson.databind.JsonNode;
import java.util.ArrayList;
import java.util.List;
import java.util.HashSet;
import java.util.Set;

public class JsonNodeListMerger {

    public static List<JsonNode> mergeJsonNodeLists(List<JsonNode> sourceList, List<JsonNode> destinationList) {
        Set<JsonNode> uniqueNodes = new HashSet<>(destinationList);
        List<JsonNode> mergedList = new ArrayList<>(destinationList);

        for (JsonNode node : sourceList) {
            if (!uniqueNodes.contains(node)) {
                mergedList.add(node);
                uniqueNodes.add(node);
            }
        }

        return mergedList;
    }

    // main method for testing
    public static void main(String[] args) {
        // Example usage
        // List<JsonNode> sourceList = ...
        // List<JsonNode> destinationList = ...
        // List<JsonNode> mergedList = mergeJsonNodeLists(sourceList, destinationList);
    }
}
