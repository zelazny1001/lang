package org.example;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;

import java.io.IOException;
import java.util.*;
import java.util.stream.Collectors;

public class EntityPriorityRulesEngine {

    public List<JsonNode> getPriorityNodes(List<JsonNode> entities, String rulesJSON) throws IOException {
        ObjectMapper mapper = new ObjectMapper();
        JsonNode rulesNode = mapper.readTree(rulesJSON).get("rules");

        Map<String, List<ConditionSelectionPair>> ruleMap = parseRules(rulesNode);

        return entities.stream()
                .collect(Collectors.groupingBy(e -> e.get("rawText").asText()))
                .values().stream()
                .map(group -> selectEntity(group, ruleMap))
                .filter(Objects::nonNull)
                .collect(Collectors.toList());
    }

    public List<JsonNode> getPriorityNodes(List<JsonNode> entities, JsonNode rules) {
        Map<String, List<ConditionSelectionPair>> ruleMap = parseRules(rules);

        return entities.stream()
                .collect(Collectors.groupingBy(e -> e.get("rawText").asText()))
                .values().stream()
                .map(group -> selectEntity(group, ruleMap))
                .filter(Objects::nonNull)
                .collect(Collectors.toList());
    }

    private JsonNode selectEntity(List<JsonNode> group, Map<String, List<ConditionSelectionPair>> ruleMap) {
        for (List<ConditionSelectionPair> pairs : ruleMap.values()) {
            for (ConditionSelectionPair pair : pairs) {
                for (JsonNode entity : group) {
                    if (pair.entityMatchesSelectionCriteria(entity)) {
                        return entity; // Return the first entity that matches any rule
                    }
                }
            }
        }
        return group.get(0); // Default to the first entity if no rule applies
    }

    private Map<String, List<ConditionSelectionPair>> parseRules(JsonNode rulesNode) {
        Map<String, List<ConditionSelectionPair>> ruleMap = new HashMap<>();
        rulesNode.forEach(ruleNode -> {
            JsonNode conditionNode = ruleNode.get("condition");
            JsonNode selectionNode = ruleNode.get("selection");

            ConditionSelectionPair pair = new ConditionSelectionPair(conditionNode, selectionNode);
            ruleMap.computeIfAbsent("rule", k -> new ArrayList<>()).add(pair);
        });
        return ruleMap;
    }

    static class ConditionSelectionPair {
        private final Map<String, String[]> conditions;
        private final Map<String, String> selections;

        ConditionSelectionPair(JsonNode condition, JsonNode selection) {
            this.conditions = new HashMap<>();
            this.selections = new HashMap<>();

            condition.fieldNames().forEachRemaining(key ->
                    conditions.put(key, condition.get(key).asText().split("\\|")));

            selection.fieldNames().forEachRemaining(key ->
                    selections.put(key, selection.get(key).asText()));
        }

        boolean entityMatchesSelectionCriteria(JsonNode entity) {
            for (Map.Entry<String, String[]> entry : conditions.entrySet()) {
                String key = entry.getKey();
                String[] conditionValues = entry.getValue();
                JsonNode entityValue = entity.get(key);

                if (entityValue == null || !Arrays.asList(conditionValues).contains(entityValue.asText())) {
                    return false; // Condition not met
                }
            }

            for (Map.Entry<String, String> entry : selections.entrySet()) {
                String key = entry.getKey();
                String selectionValue = entry.getValue();
                JsonNode entityValue = entity.get(key);

                if (entityValue != null && entityValue.asText().equals(selectionValue)) {
                    return true;
                }
            }

            return false;
        }
    }
}

package org.example;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ObjectNode;
import org.junit.Before;
import org.junit.Test;
import static org.junit.Assert.*;

import java.util.ArrayList;
import java.util.List;

public class EntityPriorityRulesEngineTest {

    private EntityPriorityRulesEngine yourClassInstance;
    private ObjectMapper mapper;

    private static String rulesJSON =
          "{'rules' : ["
        + "  {"
        + "    'condition' : {'entityConcept' : 'MerchantName', 'predictor' : 'lookup|crf'}, "
        + "    'selection' : {'predictor' : 'lookup'}"
        + "  },"
        + "  {"
        + "     'condition' : {'entityConcept' : 'Account|Date'},"
        + "     'selection' : {'entityConcept' : 'Account'}"
        + "  },"
        + "  {"
        + "     'condition' : {'entityConcept' : 'Number|Date'},"
        + "     'selection' : {'entityConcept' : 'Date'}"
        + "  },"
        + "  {"
        + "     'condition' : {'entityConcept' : 'Number|Account'},"
        + "     'selection' : {'entityConcept' : 'Account'}"
        + "  }"
        + "]}";

    @Before
    public void setUp() {
        yourClassInstance = new EntityPriorityRulesEngine();
        mapper = new ObjectMapper();
        rulesJSON = rulesJSON.replaceAll("'", "\"");
    }

    @Test
    public void testGetPriorityNodes1() throws Exception {
        List<JsonNode> entities = new ArrayList<>();
        entities.add(createEntityNode("MerchantName", "ruby tuesday", "crf", 14, 29));
        entities.add(createEntityNode("FromAccount", "checking account 2", "custom", 2, 7));
        entities.add(createEntityNode("Account", "savings", "common", 4, 12));
        entities.add(createEntityNode("Account", "2023", "custom", 8, 15));
        entities.add(createEntityNode("Date", "2023", "common", 8, 15));
        entities.add(createEntityNode("MerchantName", "ruby tuesday", "lookup", 10, 25));
        entities.add(createEntityNode("Date", "April 19", "common", 17, 35));

        List<JsonNode> result = yourClassInstance.getPriorityNodes(entities, rulesJSON);

        assertEquals(5, result.size());
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("checking account 2")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("savings")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("2023") && node.get("entityConcept").asText().equals("Account")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("ruby tuesday") && node.get("predictor").asText().equals("lookup")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("April 19")));
    }

    @Test
    public void testGetPriorityNodes2() throws Exception {
        List<JsonNode> entities = new ArrayList<>();
        entities.add(createEntityNode("Number", "2023", "crf", 14, 29));
        entities.add(createEntityNode("FromAccount", "checking account 2", "custom", 2, 7));
        entities.add(createEntityNode("Account", "savings", "common", 4, 12));
        entities.add(createEntityNode("Account", "2023", "custom", 8, 15));
        entities.add(createEntityNode("Date", "2023", "common", 8, 15));
        entities.add(createEntityNode("Date", "April 19", "common", 17, 35));

        List<JsonNode> result = yourClassInstance.getPriorityNodes(entities, rulesJSON);

        assertEquals(4, result.size());
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("checking account 2") && node.get("entityConcept").asText().equals("FromAccount")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("savings") && node.get("entityConcept").asText().equals("Account")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("2023") && node.get("entityConcept").asText().equals("Account")));
        assertTrue(result.stream().anyMatch(node -> node.get("rawText").asText().equals("April 19") && node.get("entityConcept").asText().equals("Date")));
    }
    private JsonNode createEntityNode(String entityConcept, String rawText, String predictor, int start, int end) {
        ObjectNode node = mapper.createObjectNode();
        node.put("entityConcept", entityConcept);
        node.put("rawText", rawText);
        node.put("predictor", predictor);
        node.put("start", start);
        node.put("end", end);
        return node;
    }
}
