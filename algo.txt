@startuml

class Candidate {
    + start_idx: int
    + start_dB: float
    + num_voice_chunks: int
    + num_non_voice_chunks: int
    + add_voice(): void
    + add_non_voice(): void
}

class Span {
    + start_idx: int
    + end_idx: int
    + start_dB: float
    + end_dB: float
    + num_voice_chunks: int
    + num_non_voice_chunks: int
}

class Metric {
    + voice_energies: list[float]
    + non_voice_energies: list[float]
    + chunk_decibels: list[float]
}

class SpanDetector {
    + sr: int
    + energy_threshold_0: float
    + energy_threshold_1: float
    + min_voice_chunks: int
    + min_non_voice_chunks: int
    + curr_idx: int
    + curr_candidate: Candidate
    + spans: list[Span]
    + metric: Metric
    + add_to_stream(audio): tuple
    + get_fft(audio): list
    + is_candidate_eligible(): bool
    + is_break_detected(): bool
}

SpanDetector "1" -- "1" Metric
SpanDetector "1" -- "*" Span
SpanDetector "0..1" -- "1" Candidate

@enduml

# ------------------ Sequence Diagram - Valid voice span detector

@startuml

actor Client
entity "SpanDetector" as SpanDetector
entity "Candidate" as Candidate
entity "Metric" as Metric
entity "Span" as Span

Client -> SpanDetector : add_to_stream(audio)
activate SpanDetector

SpanDetector -> SpanDetector : get_fft(audio)
activate SpanDetector

SpanDetector -> Metric : update(chunk_decibels)
activate Metric
deactivate Metric

alt energy >= voice_threshold
    SpanDetector -> Candidate : add_voice()
    activate Candidate
    deactivate Candidate

    SpanDetector -> Metric : update(voice_energies)
    activate Metric
    deactivate Metric

    SpanDetector -> SpanDetector : is_break_detected()
    activate SpanDetector
    SpanDetector -> SpanDetector : is_candidate_eligible()
    activate SpanDetector
    deactivate SpanDetector
    deactivate SpanDetector

    alt candidate is eligible
        SpanDetector -> Span : create new span
        activate Span
        deactivate Span
    end
else energy < voice_threshold
    SpanDetector -> Candidate : add_non_voice()
    activate Candidate
    deactivate Candidate

    SpanDetector -> Metric : update(non_voice_energies)
    activate Metric
    deactivate Metric
end

deactivate SpanDetector
@enduml
