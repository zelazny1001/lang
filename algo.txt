import javazoom.jl.decoder.Bitstream;

import javax.sound.sampled.AudioFileFormat;
import javax.sound.sampled.AudioFormat;
import javax.sound.sampled.AudioInputStream;
import javax.sound.sampled.AudioSystem;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class MP3Converter {
    public static void main(String[] args) {
        String root = "root/";
        Stromg mp3FilePath = root + "input.mp3";
        String wavFilePath = root + "output.wav";

        try (FileInputStream mp3InputStream = new FileInputStream(mp3FilePath)) {
            Bitstream bitstream = new Bitstream(mp3InputStream);
            Decoder decoder = new Decoder();
            File wavFile = new File(wavFilePath);
            FileOutputStream wavOutputStream = new FileOutputStream(wavFile);

            AudioFormat format = new AudioFormat(44100, 16, 1, true, false); // 44.1kHz, 16-bit, mono
            AudioInputStream audioInputStream = new AudioInputStream(mp3InputStream, format, AudioSystem.NOT_SPECIFIED);

            AudioSystem.write(audioInputStream, AudioFileFormat.Type.WAVE, wavFile);
            wavOutputStream.close();
        } catch (JavaLayerException | IOException e) {
            e.printStackTrace();
        }
    }
}
