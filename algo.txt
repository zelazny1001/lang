import openpyxl
import requests
import json

# Extract dialog of a person from the full dialog
def extractPersonDialog(fullDialog, person='Person1'):
    sentences = []
    chunks = fullDialog.split('#')
    for chunk in chunks:
        if chunk.startswith(person + ':'):
            cleaned_chunk = chunk[len(person)+2:].replace('\n', ' ')
            sentences.append(cleaned_chunk)
    return ' '.join(sentences)

# Function to get purpose and intent from REST API
def getPurposeAndIntent(personDialog, fullDialog):
    payload = {
        "utterance": personDialog,
        "batch": fullDialog,
        "customerBatch": personDialog
    }
    response = requests.post("http://localhost:9001/pltfm/rest/agentAssist/compose", json=payload)
    data = response.json()
    return data.get("callPurpose"), data.get("purposeIntent")

# Function to populate Sheet2 in Excel
def populateSheet(inputFilename="inputs.xlsx"):
    # Load workbook and sheet
    wb = openpyxl.load_workbook(inputFilename)
    ws = wb['Sheet2']

    # Clear existing values in columns F and G
    for row in ws.iter_rows(min_row=2, min_col=6, max_col=7, max_row=ws.max_row):
        for cell in row:
            cell.value = ' '

   # Freeze the top row
    ws.freeze_panes = 'A2'

    # Bold the font for the header row
    bold_font = openpyxl.styles.Font(bold=True)
    for cell in ws["1:1"]:
        cell.font = bold_font

    # Style settings
    calibri8 = openpyxl.styles.NamedStyle(name='Calibri8', font=openpyxl.styles.Font(name='Calibri', size=8))
    
    for row in ws.iter_rows(min_row=2, max_col=3, max_row=ws.max_row):
        dialog_cell = row[2]
        fullDialog = dialog_cell.value
        utterance = extractPersonDialog(fullDialog, 'Person1')
        batch = fullDialog
        customerBatch = utterance
        callPurpose, purposeIntent = getPurposeAndIntent(utterance, fullDialog)

        # Update Excel sheet
        purpose_cell = ws.cell(row=dialog_cell.row, column=6, value=callPurpose)
        intent_cell = ws.cell(row=dialog_cell.row, column=7, value=purposeIntent)
        
        # Apply styles
        purpose_cell.style = calibri8
        intent_cell.style = calibri8
        ws.column_dimensions['D'].width = 40
        ws.column_dimensions['F'].width = 40
        ws.column_dimensions['D'].alignment = openpyxl.styles.Alignment(wrap_text=True)
        ws.column_dimensions['F'].alignment = openpyxl.styles.Alignment(wrap_text=True)

    # Save workbook
    wb.save(inputFilename)

if __name__ == '__main__':
    populateSheet()
