import pandas as pd
import requests
import json

def orchestrate(inputFilename):
    # REST endpoint URL
    url = "http://localhost:9001/pltfm/rest/agentAssist/processChunk"

    # Sheet names following the pattern 'Call_0_1', 'Call_0_2', ..., 'Call_0_9'
    sheets = [f'Call_0_{i}' for i in range(1, 10)]

    for sheet in sheets:
        print(f"Processing worksheet: {sheet}")
        input_df = pd.read_excel(inputFilename, sheet_name=sheet)

        for index, row in input_df.iterrows():
            # Prepare JSON data for POST request
            payload = {
                "sessionID": row['sessionID'],
                "personID": row['personID'],
                "transcript": row['transcript'],
                "isEnd": row['isEnd']
            }

            # Make POST request and collect response
            response = requests.post(url, json=payload)

        print(f"Completed processing {sheet}")

    print("Completed processing all worksheets")
