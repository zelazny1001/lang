public String generateSessionTable(HashMap<String, Session> sessionMap) {
    StringBuilder table = new StringBuilder();

    // Start table and add header
    table.append("<table border='1' style='border-collapse: collapse;border: 1px dotted black;font-family: verdana; font-size: 10px;'>");
    table.append("<thead style='background-color: green; font-weight: bold;'>");
    table.append("<tr>");
    table.append("<th>session ID</th>");
    table.append("<th>chunk count</th>");
    table.append("<th>batch count</th>");
    table.append("</tr>");
    table.append("</thead>");
    table.append("<tbody>");

    // Add rows with session data
    boolean isAlternate = false;
    for (Map.Entry<String, Session> entry : sessionMap.entrySet()) {
        Session session = entry.getValue();
        table.append("<tr style='background-color: " + (isAlternate ? "lightblue" : "white") + ";'>");
        table.append("<td onclick=\"flashAndFetch('" + session.sessionID + "')\">").append(session.sessionID).append("</td>");
        table.append("<td>").append(session.chunkCount).append("</td>");
        table.append("<td>").append(session.batchCount).append("</td>");
        table.append("</tr>");

        // Toggle row color
        isAlternate = !isAlternate;
    }

    // End table
    table.append("</tbody>");
    table.append("</table>");

    // Add JavaScript function for cell click
    table.append("<script>");
    table.append("function flashAndFetch(sessionID) {");
    table.append("  var cell = event.target;");
    table.append("  var originalColor = cell.style.backgroundColor;");
    table.append("  cell.style.backgroundColor = 'green';");
    table.append("  setTimeout(function() { cell.style.backgroundColor = originalColor; }, 300);");
    table.append("  fetch('/viewSession?sessionID=' + sessionID);");
    table.append("}");
    table.append("</script>");

    return table.toString();
}

        table.append("<td>").append(chunk.getPersonID()).append("</td>");
        table.append("<td style='width: 250px;'>").append(chunk.getTranscript()).append("</td>");

        // Assuming there's a direct relationship between Chunk and AAResponse, 
        // such as they are in the same order or they have a shared ID.
        AAResponse response = sessionDetail.getResponses().get(sessionDetail.getChunkList().indexOf(chunk));

}
